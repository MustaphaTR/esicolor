BALTIC_save_communist_benefactor = {
	hidden_effect = {
		if = { # Soviet Majors
			limit = {
				any_other_country = {
					has_government = communism
					state_originally_from_SOV = yes
					is_major = yes
				}
			}

			every_other_country = {
				if = {
					limit = {
						has_government = communism
						state_originally_from_SOV = yes
						is_major = yes
						OR = {
							check_variable = { ROOT.communist_benefactor = 0 }
							capital_scope = { set_temp_variable = { distance_to_ROOT_capital = distance_to@ROOT.capital_scope } }
							ROOT = {
								var:communist_benefactor = {
									capital_scope = {
										check_variable = { distance_to_ROOT_capital < distance_to@ROOT.capital_scope }
									}
								}
							}
						}
					}

					ROOT = { set_variable = { communist_benefactor = PREV } }
				}
			}
		}
		else_if = { # Non-Soviet Majors
			limit = {
				any_other_country = {
					has_government = communism
					is_major = yes
				}
			}

			every_other_country = {
				if = {
					limit = {
						has_government = communism
						is_major = yes
						OR = {
							check_variable = { ROOT.communist_benefactor = 0 }
							capital_scope = { set_temp_variable = { distance_to_ROOT_capital = distance_to@ROOT.capital_scope } }
							ROOT = {
								var:communist_benefactor = {
									capital_scope = {
										check_variable = { distance_to_ROOT_capital < distance_to@ROOT.capital_scope }
									}
								}
							}
						}
					}

					ROOT = { set_variable = { communist_benefactor = PREV } }
				}
			}
		}
		else_if = { # Other Commies
			limit = {
				any_other_country = {
					has_government = communism
				}
			}

			every_other_country = {
				if = {
					limit = {
						has_government = communism
						OR = {
							check_variable = { ROOT.communist_benefactor = 0 }
							capital_scope = { set_temp_variable = { distance_to_ROOT_capital = distance_to@ROOT.capital_scope } }
							ROOT = {
								var:communist_benefactor = {
									capital_scope = {
										check_variable = { distance_to_ROOT_capital < distance_to@ROOT.capital_scope }
									}
								}
							}
						}
					}

					ROOT = { set_variable = { communist_benefactor = PREV } }
				}
			}
		}
	}
}
