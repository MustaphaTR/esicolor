	name = JAP_hitachi_arms_department_organization
	include = esicolor_generic_assault_guns_organization
	icon = GFX_idea_JAP_hitachi_arms

	allowed = {
		has_dlc = "Arms Against Tyranny"
	}
	visible = {
		owner = {
			NOT = { has_country_flag = assault_gun_manufacturer_disabled }
			OR = {
				original_tag = ZBZ
				AND = {
					has_global_flag = custom_designers_only
					owns_state = 282
				}
			}
		}
	}
	available = {
		if = {
			limit = { has_global_flag = custom_designers_only }

			owner = {
				282 = { state_designer_availabile = yes }
			}
		}
	}
}
