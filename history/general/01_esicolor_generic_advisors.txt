# For minor countries where advisors are not defined
# Generate them from a generic template

every_possible_country = {
	limit = { NOT = { original_tag = OOO } }

	if = {
		limit = { NOT = { has_variable = original_capital } }

		set_variable = { original_capital = 1 }
	}

	# Ideology Advisors
	if = {
		limit = {
			NOT = {
				has_allowed_idea_with_traits = { idea = staunch_monarchist characters = yes }
			}
		}

		meta_effect = {
			text = {
				generate_character = {
					token_base = [TAG]_generic_staunch_monarchist
					advisor = {
						slot = political_advisor
						traits = { staunch_monarchist }

						visible = {
							OR = {
								NOT = { has_global_flag = custom_advisors_only }
								has_character_flag = special_advisor
							}
						}
						available = {
							if = {
								limit = { has_focus_tree = horn_of_africa_tree }

								OR = {
									has_government = neutrality
									has_completed_focus = HOA_a_change_in_course
								}
							}
							hidden_trigger = {
								OR = {
									NOT = { has_global_flag = custom_advisors_only }
									has_character_flag = special_advisor
								}
							}
							if = {
								limit = { has_global_flag = custom_advisors_only_core }

								[STATE] = { is_core_of =  ROOT }
							}
						}

						ai_will_do = {
							factor = 0

							modifier = {
								has_country_flag = ai_go_neutrality
								NOT = { has_government = neutrality }

								add = 30
							}
						}
					}
					portraits = {
						army = { small = GFX_idea_unknown_advisor }
					}
				}
			}
			TAG = "[THIS.GetTag]"
			STATE = "[?THIS.original_capital.GetID]"
		}
	}
}
